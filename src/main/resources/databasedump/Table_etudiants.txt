CREATE TABLE ETUDIANT (
	ID_ETUDIANT INT NOT NULL AUTO_INCREMENT,
	NOM_ETUDIANT VARCHAR(255) NOT NULL,
	PRENOM_ETUDIANT VARCHAR(255) NOT NULL,
	MAIL_ETUDIANT VARCHAR(255) NULL,
	CONVENTION_ID_CONVENTION INT NOT NULL,
	PRIMARY KEY (ID_ETUDIANT),
	CONSTRAINT FK_CONVENTION_ID_CONVENTION
		FOREIGN KEY CONVENTION_ID_CONVENTION
		REFERENCES CONVENTION(ID_CONVENTION)
) ENGINE InnoDB DEFAULT CHARSET=utf8;
